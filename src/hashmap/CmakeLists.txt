# Hash map library
add_library(hashmap "")

target_sources(
    hashmap
    PRIVATE
        HashMap.cpp
    PUBLIC
        ${CMAKE_CURRENT_LIST_DIR}/HashMap.h
    )

target_include_directories(
    hashmap
    PUBLIC
        ${CMAKE_CURRENT_LIST_DIR}
    )

# Main executable
set(SRCS main.cpp)

add_executable(main-hashmap ${SRCS})
target_link_libraries(main-hashmap hashmap)
